function __call( $class, $args ) {
  if(array_key_exists( $class, $this->relationships )) {
    if( $this->relationships[ $class ] == 'has_a') {
      if ( isset( $this->{$class} ) ) {
        return $this->{$class}->{$args[0]};
      }
      else {
        $this->$class  = new $class( $this->id );
        return  $this->{$class}->{$args[0]};
      }
    }
  }
}
