will not work in your situation:

function tst_func ($statement, &$test1) {
	if ( !isset ($test1) ) {
		$test1 = "the usual";
	}
	echo $statement . $test1 . "\n";
}

$tst_str  = "something new";
$tst_ref = & $tst_str;

tst_func( "Stuff was: ",  $tst_ref );
tst_func( "Stuff was: " );
