> function http_build_query($formdata,$numeric_prefix = '',$arg_separator
> = null,$_parent = null) {
>     if (empty($arg_separator)) {
>         $arg_separator = ini_get('arg_separator.output');
>     }
> 
>     $postdata = array();
>     foreach ($formdata as $k => $v) {
>         if (!empty($_parent)) {
>             $k = $_parent .'['. urlencode($k) .']';
>         } else {
>             if (is_numeric($k)) {
>                 $k = $numeric_prefix . $k;
>             }
>             $k = urlencode($k);
>         }
>         if (is_array($v)) {
>             $postdata[] =

>         } else {
>             $postdata[] = $k .'=' . urlencode($v);
>         }
>     }
>

> }
