<?php

class Bean_Test extends PHP_Bean {
    var $values = array ();

    function Bean_Test (&$server) {
        $this->init ($server);
        $this->namespace = 'test';
        $this->addMethods (__FILE__);
    }

    /**
     * Sets a named value.
     * @access public
     * @param string
     * @param string
     */
    function set ($name, $value) {
        $this->values[$name] = $value;
    }

    /**
     * Gets a named value.
     * @access public
     * @param string
     * @return string
     */
    function get ($name) {
        return $this->values[$name];
    }
}
