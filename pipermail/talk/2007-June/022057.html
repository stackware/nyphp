  <input value="Display Value" onfocus="clearText (this);" /> is a bit more complete.

   

  Or <input value="Display Value" onfocus="clearText (this);" onblur="restoreText (this);" /> can replace empty values with your label.

   

  <script type="text/javascript">

    function clearText (el) {

      if (el.defaultValue == el.value) el.value = "";

    }

    function restoreText (el) {

      if (el.value == "") el.value = el.defaultValue;

    }

  </script>
