You can try using the call_user_func function here:

<?php
class MyClass {
    function setMaxLength($max, $errorMsg) {
        echo $errorMsg.' ('.$max.')';
    }
}


$myObj = new MyClass();

call_user_func(array($myObj,'setMaxLength'), 40, 'The last name cannot be
more than 40 characters');
?>
