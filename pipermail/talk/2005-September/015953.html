quick and easy

 *function* openUrl($url) {
       *// Fake the browser type*
       *// ini_set('user_agent','MSIE 4\.0b2;');*
    $handle = fopen($url, "rb");
    $contents = '';
    if($handle) {
        while (!feof($handle)) {
            $contents .= fread($handle, 8192);
        }
        fclose($handle);       
    }
    else {
        $contents = 'Error opening $url' . $url;
    }
    return $contents;
} 



print $contents = openUrl('http://www.google.com');
