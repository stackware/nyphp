You must set the value on the input.

 while( $row = mysql_fetch_array( $category ) )
{
 echo(
    "<tr><td>" .$row['expenses_cat_desc'] ."</td><td><input type=text
name=expense_cat value=" . $row['expenses_cat_number'] . " /></td></tr>");
}
