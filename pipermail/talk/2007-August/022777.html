For future reference, the easiest way (in postDispatch()) was:

$dbAdapter = Zend_Registry::get( 'dbAdapter' );
$profiler = $dbAdapter->getProfiler( );
$query = $profiler->getLastQueryProfile( );
echo $query->getQuery( );
