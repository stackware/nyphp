Tell me about it.  Here's what we use:

function replace_fancy_characters($in) {
    static $search, $replace;

    if (isset($search) == false) {
        $search = array(
            '/\x96/',
            '/\x97/',
            '/\x91/',
            '/\x92/',
            '/\x93/',
            '/\x94/',
            '/\x85/',
            '/\x95/',
            '/\x09/',
            '/\xBC/',
            '/\xBD/',
            '/\xBE/',
        );

        $replace = array(
            '-',
            '--',
            "'",
            "'",
            '"',
            '"',
            '...',
            '*',
            ' ',
            '1/4',
            '1/2',
            '3/4',
        );
    }
    return preg_replace($search, $replace, $in);
}
