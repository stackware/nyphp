The syntax should be:

for ($x = 0; $x < 10; $x++) {
   switch (TRUE) {
     case ($x < 3 || $x > 6):
       echo "x = $x: yes!\n";
       break;
     default:
       echo "x = $x: no!\n";
   }
}
