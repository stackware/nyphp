SELECT
SUM( IF ( win = $team, 1, 0 ) )  AS w,
SUM( IF ( loss = $team, 1, 0 ) )  AS l
FROM games

>
> --------------
> | win | loss |
> --------------
> |   1 |    2 |
> |   1 |    3 |
> |   2 |    3 |

> can do this with two queries, like so:
