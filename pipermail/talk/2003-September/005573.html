> > So, a quick and dirty example...
> >
> >
> > if ( empty($_SERVER['argv'][1]) ) {
> >     $_SERVER['argv'][1] = '';

> >     case 'blah':
> >         $In = fopen('php://stdin', 'r');
> >
> >         while ( !feof($In) ) {

> >         }
> >         break;

> >         echo 'usage...';
