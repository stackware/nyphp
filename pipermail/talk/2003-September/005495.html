  This should be enough to get you started: 

  $fp = fopen('foo.tab', 'r'); 
  print "<table>\n"; 
  while ($line = fgetcsv($fp, 1000, "\t")) { 
    print "  <tr>\n    <td>"; 
    print join("</td>\n    <td>", $line); 
    print "</td>\n  </tr>\n"; 
  } 
  print "</table>\n"; 
  fclose($fp);
