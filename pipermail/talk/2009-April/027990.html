DRY! DRY! DRY!

Try something like this:

$cart_keys = array(
'mouse_choice','monitor_choice','wifi_choice','subtotal','offsets','amount','desc','first_name','last_name','address','city','state','zip','phone','billeve','email','ship_to_first_name','ship_to_last_name','ship_to_address','ship_to_city','ship_to_state','ship_to_zip','shipday','card_num','exp_date','card_code');

foreach($cart_keys as $key) {
  if($_POST[$key])
    $_SESSION[$key] = $_POST[$key];
}

// load the keys as variables, not really needed because the values
already exist in  _SESSION.
extract(array_intersect_key($_SESSION,array_flip($cart_keys)));
