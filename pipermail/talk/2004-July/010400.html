what about something like this:

$all_teams = array(
	'TEAM 1' => array(
		'Year 1' => 8,
		'Year 2' => 9,
		'Year 3' => 22
	),
	'TEAM 2' => array(
		'Year 1' => 18,
		'Year 2' => 94,
		'Year 3' => 12
	),
	etc...
);

$year1 = 0;
$year2 = 0;
$year3 = 0;

foreach ($all_teams as $next_array)
{
	$year1 += $next_array['Year 1'];
	$year2 += $next_array['Year 2'];
	$year3 += $next_array['Year 3'];
}
