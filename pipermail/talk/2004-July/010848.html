>>>>>Actually you would be better off with:

>>>>>function setPart($part = NULL, $val = NULL, $append = FALSE) {
>>>>>  if (isset($part) && isset($val) && @isset($this->{'_'.$part})) {

>>>>>      $this->{'_'.$part} .= $val;
