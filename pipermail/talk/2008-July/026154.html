Note: This response is more or less tongue in cheek, and is not meant
to be taken seriously.

<?php
$y="tnk81|98.8$|yuhj78t|32.6$|tris78y|459.78$|";

$y_len = strlen($y);

$results = array();
$in_name = true;

$name = '';
$value = '';

for ($i=0; $i <= $y_len; $i++) {
    if ($y[$i] == '|') {
        if (!$in_name) {
            if ($name) {
                $results[$name] = $value;
                $name = '';
                $value = '';
            }
        }
        $in_name = !$in_name;
    } else {
        if ($in_name) {
            $name .= $y[$i];
        } else {
            $value .= $y[$i];
        }
    }
}
var_dump($results);

  string(5) "32.6$"

}
