This works (tested)

<?php
$y="tnk81|98.8$|yuhj78t|32.6$|tris78y|459.78$";
$Parts = explode("|",$y);
$Count = count($Parts);
for($Sub1=0;$Sub1<=$Count;$Sub1 += 2) {
   $i = $Sub1 + 1;
   echo "<br />$Parts[$Sub1] $Parts[$i]\n";
   }
?>
