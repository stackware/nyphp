              //Sort on 'Dep_Name'
              foreach ($rows as $val) 
	      {
                $sortarray[] = $val['Dept_Name'];
              }
              
              //this will osrt your multi associative aray by the
              //column heading you specified
              array_multisort($sortarray, $rows);
