I use this function:

function validateEmail(email)
{
        var
filter=/^(\w+(?:\.\w+)*)@((?:\w+\.)*\w[\w-]{0,66})\.([a-z]{2,6}(?:\.[a-z
]{2})?)$/i;
        if(filter.test(email)==true) {
                return true;
        } else{
                return false;
        }

}
